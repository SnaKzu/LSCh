# ===================================================================
# NOTAS DE INSTALACIÓN
# ===================================================================
#
# 1. CREAR ENTORNO VIRTUAL (Recomendado):
#    python -m venv venv
#    .\venv\Scripts\Activate.ps1  # Windows PowerShell
#    source venv/bin/activate      # Linux/Mac
#
# 2. INSTALAR DEPENDENCIAS:
#    pip install --upgrade pip
#    pip install -r requirements.txt
#
# 3. VERIFICAR INSTALACIÓN:
#    python -c "import tensorflow as tf; print(f'TensorFlow: {tf.__version__}')"
#    python -c "import mediapipe; print('MediaPipe: OK')"
#    python -c "from config_manager import config; print('Config: OK')"
#
# 4. PROBLEMAS COMUNES:
#
#    a) Error con protobuf:
#       pip install protobuf==3.20.3 --force-reinstall
#
#    b) Error con MediaPipe en Python 3.12:
#       Usar Python 3.10 o 3.11 (MediaPipe tiene mejor soporte)
#
#    c) Error con TensorFlow en GPU:
#       pip install tensorflow[and-cuda]==2.15.0
#
#    d) Error con PyQt5:
#       Instalar Qt5: https://www.qt.io/download
#
#    e) Conflictos de versiones:
#       pip install --upgrade --force-reinstall -r requirements.txt
#
# 5. PARA CONTRIBUIDORES:
#    pip install -e .
#    pip install -r requirements.txt
